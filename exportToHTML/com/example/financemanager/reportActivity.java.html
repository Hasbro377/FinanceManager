124      // Новый метод для Stacked Bar Chart
125      private void showStackedBarChart() {
126          ArrayList<BarEntry> entries = new ArrayList<>();
127          ArrayList<String> labels = new ArrayList<>();
128  
129          // Получаем данные по категориям
130          List<String> categories = dbHelper.getCategoriesForWallet(selectedWallet, userID);
131          if (categories.isEmpty()) {
132              barChartStacked.clear();  // Очищаем график, если нет данных
133              barChartStacked.setNoDataText("Нет данных для отображения");
134              barChartStacked.invalidate();
135              return;
136          }
137  
138          for (int i = 0; i < categories.size(); i++) {
139              String category = categories.get(i);
140              float income = dbHelper.getSumForCategoryAndType("Доход", category, selectedWallet, userID);
141              float expense = dbHelper.getSumForCategoryAndType("Расход", category, selectedWallet, userID);
142              entries.add(new BarEntry(i, new float[]{expense, income}));
143              labels.add(category);  // Добавляем название категории
144          }
145  
146          BarDataSet dataSet = new BarDataSet(entries, "");
147          dataSet.setColors(new int[]{Color.RED, Color.GREEN});
148          dataSet.setStackLabels(new String[]{"Расходы", "Доходы"});
149  
150          BarData barData = new BarData(dataSet);
151          barChartStacked.setData(barData);
152  
153          // Настройка горизонтальной оси (XAxis)
154          XAxis xAxis = barChartStacked.getXAxis();
155          xAxis.setValueFormatter(new IndexAxisValueFormatter(labels));  // Названия категорий
156          xAxis.setPosition(XAxis.XAxisPosition.BOTTOM);
157          xAxis.setGranularity(1f);  // Шаг между метками
158          xAxis.setGranularityEnabled(true);
159          xAxis.setLabelCount(labels.size());  // Количество меток = количеству категорий
160          xAxis.setTextSize(12f);  // Размер текста
161          xAxis.setTextColor(Color.BLACK);
162          xAxis.setDrawGridLines(false);  // Убираем сетку
163          xAxis.setLabelRotationAngle(-45f);  // Поворот текста для длинных названий
164  
165  
166          // Настройка вертикальной оси (YAxis)
167          YAxis leftAxis = barChartStacked.getAxisLeft();
168          leftAxis.setTextSize(12f);
169          leftAxis.setTextColor(Color.BLACK);
170          barChartStacked.getAxisRight().setEnabled(false);  // Отключаем правую ось
171  
172          barChartStacked.getDescription().setEnabled(false);
173          barChartStacked.setFitBars(true);  // Подгонка столбцов
174          barChartStacked.invalidate();  // Обновление графика
175      }

228      private void setTextDate() {
229          final Calendar calendar = Calendar.getInstance();
230          int year = calendar.get(Calendar.YEAR);
231          int month = calendar.get(Calendar.MONTH);
232          int day = calendar.get(Calendar.DAY_OF_MONTH);
233          String selectedDate = day + "." + (month + 1) + "." + year;
234          textBeforeDate.setText(selectedDate);
235          day = 1;
236          selectedDate = day + "." + (month + 1) + "." + year;
237          textFromDate.setText(selectedDate);
238      }

240      public void setFromDate(View view) {
241          String dateString = textFromDate.getText().toString();
242          String[] dateParts = dateString.split("\\.");
243  
244          int day = Integer.parseInt(dateParts[0]);
245          int month = Integer.parseInt(dateParts[1]);
246          int year = Integer.parseInt(dateParts[2]);
247  
248          DatePickerDialog datePickerDialog = new DatePickerDialog(this,
249                  new DatePickerDialog.OnDateSetListener() {
250                      @Override
251                      public void onDateSet(DatePicker view, int selectedYear, int selectedMonth, int selectedDay) {
252                          String selectedDate = selectedDay + "." + (selectedMonth + 1) + "." + selectedYear;
253                          textFromDate.setText(selectedDate);
254                          updateReportData();  // Обновляем отчет при изменении даты
255                      }
256                  }, year, (month - 1), day);
257          datePickerDialog.show();
258      }
