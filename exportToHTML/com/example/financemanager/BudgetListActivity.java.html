1    package com.example.financemanager;
2    
3    import android.content.Intent;
4    import android.database.Cursor;
5    import android.database.sqlite.SQLiteDatabase;
6    import android.os.Bundle;
7    import android.widget.Button;
8    
9    import androidx.recyclerview.widget.LinearLayoutManager;
10   import androidx.recyclerview.widget.RecyclerView;
11   
12   import com.google.android.material.tabs.TabLayout;
13   
14   import java.util.ArrayList;
15   import java.util.List;
16   
17   public class BudgetListActivity extends BaseActivity {
18   
19       private RecyclerView recyclerViewBudgets;
20       private Button buttonAddBudget;
21       private TabLayout tabLayoutWallets;
22       private BudgetAdapter budgetAdapter;
23       private int userId;
24       private List<Integer> walletIds = new ArrayList<>();  // Список ID кошельков для табов
25   
26       @Override
27       protected void onCreate(Bundle savedInstanceState) {
28           super.onCreate(savedInstanceState);
29           setContentViewInFrame(R.layout.activity_budget_list);
30   
31           userId = UserSession.getCurrentUserId();
32   
33           recyclerViewBudgets = findViewById(R.id.recyclerViewBudgets);
34           buttonAddBudget = findViewById(R.id.buttonAddBudget);
35           tabLayoutWallets = findViewById(R.id.tab_layout_wallets);
36   
37           recyclerViewBudgets.setLayoutManager(new LinearLayoutManager(this));
38           budgetAdapter = new BudgetAdapter(this, userId, -1);  // -1 для всех кошельков по умолчанию
39           recyclerViewBudgets.setAdapter(budgetAdapter);
40   
41           loadWalletsIntoTabs();  // Загружаем кошельки в TabLayout
42   
43           // Обработчик выбора таба
44           tabLayoutWallets.addOnTabSelectedListener(new TabLayout.OnTabSelectedListener() {
45               @Override
46               public void onTabSelected(TabLayout.Tab tab) {
47                   int position = tab.getPosition();
48                   int selectedWalletId = (position == 0) ? -1 : walletIds.get(position - 1);  // 0 - "Все", остальные - ID
49                   budgetAdapter.updateWalletFilter(selectedWalletId);
50               }
51   
52               @Override
53               public void onTabUnselected(TabLayout.Tab tab) {}
54   
55               @Override
56               public void onTabReselected(TabLayout.Tab tab) {}
57           });
58   
59           buttonAddBudget.setOnClickListener(v -> {
60               Intent intent = new Intent(this, AddEditBudgetActivity.class);
61               intent.putExtra("budget_id", -1);
62               startActivity(intent);
63           });
64       }
65   
66       private void loadWalletsIntoTabs() {
67           SQLiteDatabase db = new DBhelper(this).getReadableDatabase();
68           Cursor cursor = db.rawQuery("SELECT id, name FROM Wallets WHERE user_id = ?", new String[]{String.valueOf(userId)});
69           tabLayoutWallets.addTab(tabLayoutWallets.newTab().setText("Все"));  // Первый таб - все кошельки
70           while (cursor.moveToNext()) {
71               walletIds.add(cursor.getInt(0));
72               tabLayoutWallets.addTab(tabLayoutWallets.newTab().setText(cursor.getString(1)));
73           }
74           cursor.close();
75           db.close();
76       }
77   
78       @Override
79       protected void onResume() {
80           super.onResume();
81           budgetAdapter.loadBudgets();  // Обновляем список
82       }
83   }
