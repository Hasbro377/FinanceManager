1    package com.example.financemanager;
2    
3    import android.content.Intent;
4    import android.os.Bundle;
5    import android.widget.FrameLayout;
6    
7    import androidx.appcompat.app.ActionBarDrawerToggle;
8    import androidx.appcompat.app.AppCompatActivity;
9    import androidx.drawerlayout.widget.DrawerLayout;
10   import com.google.android.material.navigation.NavigationView;
11   import androidx.appcompat.widget.Toolbar;
12   
13   
14   public class BaseActivity extends AppCompatActivity {
15       protected DrawerLayout drawerLayout;
16       protected NavigationView navigationView;
17       protected Toolbar toolbar;
18   
19       @Override
20       protected void onCreate(Bundle savedInstanceState) {
21           super.onCreate(savedInstanceState);
22           setContentView(R.layout.activity_base);  // Устанавливаем общий layout
23   
24           // Настройка Toolbar
25           toolbar = findViewById(R.id.toolbar);
26           setSupportActionBar(toolbar);
27   
28           // Настройка Drawer
29           drawerLayout = findViewById(R.id.drawer_layout);
30           navigationView = findViewById(R.id.nav_view);
31   
32           // Toggle для открытия/закрытия Drawer (гамбургер-иконка)
33           ActionBarDrawerToggle toggle = new ActionBarDrawerToggle(
34                   this, drawerLayout, toolbar, R.string.open_drawer, R.string.close_drawer);
35           drawerLayout.addDrawerListener(toggle);
36           toggle.syncState();
37   
38           // Обработчик кликов по пунктам меню
39           navigationView.setNavigationItemSelectedListener(item -> {
40               int id = item.getItemId();
41               if (id == R.id.nav_home) {
42                   // Переход к экрану кошельков
43                   startActivity(new Intent(this, startActivity.class));
44               } else if (id == R.id.nav_report) {
45                   // Переход к отчетам
46                   startActivity(new Intent(this, reportActivity.class));
47               } else if (id == R.id.nav_budget) {
48                   // Переход к бюджетированию
49                   startActivity(new Intent(this, BudgetListActivity.class));
50               }
51               drawerLayout.closeDrawers();  // Закрываем Drawer после выбора
52               return true;
53           });
54       }
55   
56       // Метод для установки контента в FrameLayout (вызывайте в дочерних Activity)
57       protected void setContentViewInFrame(int layoutResID) {
58           FrameLayout contentFrame = findViewById(R.id.content_frame);
59           getLayoutInflater().inflate(layoutResID, contentFrame, true);
60       }
61   }
