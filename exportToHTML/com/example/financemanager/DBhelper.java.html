129      public String getWalletBalanceAndCurrency(int userId, String walletName) {
130          String result = "0 USD";  // Значение по умолчанию
131          SQLiteDatabase db = this.getReadableDatabase();  // Открываем базу для чтения
132  
133          String query = "SELECT balance, currency FROM Wallets WHERE name = ? AND user_id = ?";
134          Cursor cursor = db.rawQuery(query, new String[]{walletName, String.valueOf(userId)});
135  
136          if (cursor != null) {
137              try {
138                  if (cursor.moveToFirst()) {  // Проверяем, есть ли результаты
139                      int balanceIndex = cursor.getColumnIndex("balance");
140                      int currencyIndex = cursor.getColumnIndex("currency");
141  
142                      // Проверка на -1
143                      if (balanceIndex == -1 || currencyIndex == -1) {
144                          Log.e("DB", "Столбцы не найдены: balanceIndex=" + balanceIndex + ", currencyIndex=" + currencyIndex);
145                          return result;  // Возвращаем значение по умолчанию
146                      }
147  
148                      double balance = cursor.getDouble(balanceIndex);
149                      String currency = cursor.getString(currencyIndex);
150                      result = String.format("%.2f %s", balance, currency);  // Форматируем строку
151                  }
152              } finally {
153                  cursor.close();  // Закрываем курсор
154              }
155          }
156          db.close();  // Закрываем базу
157          return result;  // Возвращаем результат
158      }

342          ArrayList<PieEntry> entries = new ArrayList<>();
343          SQLiteDatabase db = this.getReadableDatabase();
344  
345          // Запрос для извлечения данных
346          String query = "SELECT c.name, SUM(t.summ) as total " +
347                  "FROM Transactions t " +
348                  "JOIN Category c ON t.category_id = c.id " +
349                  "JOIN Wallets w ON t.wallet_id = w.id " +
350                  "WHERE c.type = ? AND w.name = ? " +
351                  "GROUP BY c.name";
352  
353          Cursor cursor = db.rawQuery(query, new String[]{type, walletName});
354  
355          if (cursor.moveToFirst()) {
356              int nameIndex = cursor.getColumnIndex("name");
357              int sumIndex = cursor.getColumnIndex("total");
358              do {
359                  String categoryName = cursor.getString(nameIndex);
360                  float totalSum = cursor.getFloat(sumIndex);
361  
362                  entries.add(new PieEntry(totalSum, categoryName));
363              } while (cursor.moveToNext());
364          }
365          cursor.close();
366          return entries;
367      }
