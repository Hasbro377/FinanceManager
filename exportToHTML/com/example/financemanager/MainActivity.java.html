108      public void saveData(View view) {                       //кнопка Сохранить
109          String summ = textPrice.getText().toString();       //записываем данные из текстовых элементов
110          String date = textDate.getText().toString();
111          String description = textDescription.getText().toString();
112          String selectedWallet = spinWallet.getSelectedItem().toString();
113          String selectedCategory = spinCategory.getSelectedItem().toString();
114  
115          String text = mathSumm();
116          if (text == "-1") {
117              return;
118          }else summ=text;
119          summ = summ.replace(',', '.'); // Заменяем запятую на точку
120  
121          if (!summ.isEmpty() && !date.isEmpty()) {
122              Boolean checkInsertData = dbHelper.insertTransactions(userID, selectedWallet, selectedCategory, summ, date, description);  //вызываем созданный метод
123  
124              if (checkInsertData) Toast.makeText(this, "Хорошо. Значения добавлены.", Toast.LENGTH_SHORT).show();    //вызываем всплывающее окно
125              else Toast.makeText(this, "Ошибка. Значения НЕ добавлены", Toast.LENGTH_SHORT).show();
126  
127              dbHelper.updateBudgetProgress(userID,selectedWallet, selectedCategory, summ);  // amount — сумма расхода
128  
129              dbHelper.updateWalletBalanceByName(userID, selectedWallet);                         //обновление баланса кошелька
130              Intent intent = new Intent(this, startActivity.class);          //запускаем главное Activity
131              intent.putExtra("spinnerIndex", spinWallet.getSelectedItemPosition()); // Передача индекса
132              startActivity(intent);
133          }
134          else Toast.makeText(this, "Ошибка. Запишите значения", Toast.LENGTH_SHORT).show();
135      }
136      public String mathSumm (){
137  
138          Expression e = new Expression();    //объявление и создание объекта е для решения математических задач
139          String text = textPrice.getText().toString();
140          e.setExpressionString(text);            // используем метод для РЕШЕНИЕ задачи
141          double answer = e.calculate();
142  
143          if (answer <= 0 || answer >= 9999999 || Double.toString(answer) == "NaN"){
144              Toast.makeText(this, "Ошибка. Некорректная сумма", Toast.LENGTH_SHORT).show();
145              return "-1";
146          }
147          DecimalFormat df = new DecimalFormat("#.##");
148          text = df.format(answer);
149          if (text.substring((text.length() - 2)).equals(".0")) text = text.substring(0,text.length()-2); //убирает .0 в конце целого числа
150          Log.d("Log.m", String.valueOf(answer));
151          return text;
152      }

253                          textDate.setText(selectedDate);     // Обновляем TextView с выбранной датой
