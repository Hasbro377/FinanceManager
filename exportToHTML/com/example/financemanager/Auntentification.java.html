21   public class Auntentification extends AppCompatActivity {
22       DBhelper dbHelper;              //объект для работы с БД
23       Button buttonLogin;
24       TextView textViewRegister;
25       EditText editTextEmail, editTextPassword;
26       @SuppressLint("MissingInflatedId")
27       @Override
28       protected void onCreate(Bundle savedInstanceState) {
29           super.onCreate(savedInstanceState);
30           EdgeToEdge.enable(this);
31           setContentView(R.layout.activity_auntentification);
32           ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main), (v, insets) -> {
33               Insets systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars());

35               return insets;
36           });
37           buttonLogin = findViewById(R.id.buttonLogin);
38           textViewRegister = findViewById(R.id.textViewRegister);
39           editTextEmail = findViewById(R.id.editTextEmail);
40           editTextPassword = findViewById(R.id.editTextPassword);
41           // Создание экземпляра DatabaseHelper
42           dbHelper = new DBhelper(this);
43       }

46       public void LoginAccount(View view) {
47           String email = editTextEmail.getText().toString().trim();
48           String password = editTextPassword.getText().toString().trim();
49           int userId = dbHelper.loginUser (email, password);
50           // Вызов метода входа
51           if (userId != 0) {
52               UserSession.setCurrentUserId(userId);   // Сохраняем userid в глобальном классе
53               Toast.makeText(this, "Вход успешен", Toast.LENGTH_SHORT).show();
54               Intent intent = new Intent(this, startActivity.class);          //запускаем главное Activity
55               startActivity(intent);
56               finish(); // Закрываем текущую активность
57           } else {
58               Toast.makeText(this, "Ошибка входа", Toast.LENGTH_SHORT).show();
59           }
60       }
61   
62       public void showRegistrationDialog(View view) {
63           AlertDialog.Builder builder = new AlertDialog.Builder(this);
64           builder.setTitle("Регистрация");
65   
66           // Создание пользовательского интерфейса для диалога регистрации
67           View dialogView = LayoutInflater.from(this).inflate(R.layout.dialog_registration, null);
68           builder.setView(dialogView);
69   
70           final EditText editTextRegEmail = dialogView.findViewById(R.id.editTextRegEmail);
71           final EditText editTextRegPassword = dialogView.findViewById(R.id.editTextRegPassword);
72   
73           builder.setPositiveButton("Сохранить", new DialogInterface.OnClickListener() {
74               @Override
75               public void onClick(DialogInterface dialog, int which) {
76                   String email = editTextRegEmail.getText().toString().trim();
77                   String password = editTextRegPassword.getText().toString().trim();
78                   // Логика регистрации
79                   if (dbHelper.registerUser (email, password)) {
80                       Toast.makeText(Auntentification.this, "Регистрация успешна", Toast.LENGTH_SHORT).show();
81                   } else {
82                       Toast.makeText(Auntentification.this, "Ошибка регистрации", Toast.LENGTH_SHORT).show();
83                   }
84               }
85           });
86   
87           builder.setNegativeButton("Отмена", new DialogInterface.OnClickListener() {
88               @Override
89               public void onClick(DialogInterface dialog, int which) {
90                   dialog.dismiss();
91               }
92           });
93   
94           builder.create().show();
95       }
