30           this.context = context;                                                 //конструктор для определения данных
31           this.date_id = date_id;
32           this.category_id = category_id;
33           this.summ_id = summ_id;
34           this.description_id = description_id;

39       @NonNull
40       @Override
41       public MyViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) {   //создает представление для вывода данных, само индивидуальное окно
42           View v = LayoutInflater.from(context).inflate(R.layout.userentry,parent,false); //метод возвращает объект View из файла XML-разметки, макет для записей
43           return new MyViewHolder(v);                                                     //отправляем объект в метод для связи данных и визуального жлемента
44       }
45       @Override
46       public void onBindViewHolder(@NonNull MyViewHolder holder, int position) {          //установка текста в элементе
47           holder.date_id.setText(String.valueOf(date_id.get(position)));
48           holder.category_id.setText(String.valueOf(category_id.get(position)));
49           holder.summ_id.setText(String.valueOf(summ_id.get(position)));
50           holder.description_id.setText(String.valueOf(description_id.get(position)));
51   
52           String category = (String) category_id.get(position);
53           String categoryType = dbHelper.getTypeCategory(userID, category);
54           // Устанавливаем цвет фона в зависимости от типа категории
55           if (categoryType.equals("Расход")) {
56               holder.itemView.setBackgroundColor(Color.argb(255, 255, 119, 119));
57           } else {
58               holder.itemView.setBackgroundColor(Color.argb(144, 119, 255, 119));
59           }

61   
62   
63       }

68       public class MyViewHolder extends RecyclerView.ViewHolder {
69           TextView date_id, category_id, summ_id, description_id;                         //определение используемых элементов
70           public MyViewHolder(@NonNull View itemView) {
71               super(itemView);
72               date_id = itemView.findViewById(R.id.textDate);                             //инициализация переменных для ссылки на элементы
73               category_id = itemView.findViewById(R.id.textCategory);
74               summ_id = itemView.findViewById(R.id.textSumm);
75               description_id = itemView.findViewById(R.id.textDescription);
76               itemView.setOnClickListener(new View.OnClickListener() {                    //метод для обработки нажатия на элемент в RecyclerView
77                   @SuppressLint("ResourceType")
78                   @Override
79                   public void onClick(View v) {
80                       Context context = v.getContext();                                   //объявляем Контекст
81                       Intent intent = new Intent(context, MainActivity.class);            //объявляем класс для запуска нового Activity
82                       String [] editElements = new String[3];                             //массив для хранения данных для передачи
83                       editElements[0] = date_id.getText().toString();                      // Дата
84                       editElements[1] = category_id.getText().toString();                  // Категория
85                       editElements[2] = summ_id.getText().toString();                      // Сумма
86   
87                       intent.putExtra("spinnerIndex", spinWallet.getSelectedItemPosition());  //передаем пункт активного Кошелка из раскрывающегося списка
88                       intent.putExtra("allInfo", editElements);                  //передаем данне
89                       context.startActivity(intent);                                          //открываем Activity
90                   }
91               });
92           }
93       }
